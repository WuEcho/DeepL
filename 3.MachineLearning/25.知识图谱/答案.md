```go
/*
*
有一个已经排好序的整数序列(升序， 无重复项) ，序列中可能有正整数、负整数或者 0，请用你认为 最
优的方法求序列中绝对值最小的数。
可以用任何语言实现。
输入：一个有序的整数序列。
输出：绝对值最小的数
*
*/
package main

import (
	"math"
)

func findMinAbs(nums []int) int {
	left, right := 0, len(nums)-1
	if nums[0] > 0 {
		return nums[0]
	}
	if nums[len(nums)-1] <= 0 {
		return nums[len(nums)-1]
	}
	for left <= right {
		mid := (left + right) / 2
		if nums[mid] == 0 {
			return 0
		} else if nums[mid] < 0 {
			left = mid + 1
		} else {
			right = mid - 1
		}
	}
	if math.Abs(float64(nums[right])) <= math.Abs(float64(nums[left])) {
		return nums[right]
	}
	return nums[left]
}

func main() {
	a := []int{-10, -3, -1, 2, 4, 5}
	result := findMinAbs(a)
	println("%v", result)
}
```
